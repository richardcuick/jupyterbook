double alpha = 0.05;
int n = numbers.Count;

double s = 0.0;
for (int k = 0; k < n; k++)
{
    for (int j = k + 1; j < n; j++)
    {
        s += Math.Sign(numbers[j] - numbers[k]);
    }
}

Dictionary<double, long> unique = new Dictionary<double, long>();
foreach (var number in numbers)
{
    if (unique.ContainsKey(number))
    {
        unique[number]++;
    }
    else
    {
        unique[number] = 1;
    }
}

int g = unique.Count;
Console.WriteLine("g: " + g);

double vars;
if (g == n)
{
    vars = (n * (n - 1) * (2 * n + 5)) / 18.0;
}
else
{
    long tpsum = unique.Values.Sum(tp => tp * (tp - 1) * (2 * tp + 5));
    Console.WriteLine("tpsum(freq): " + tpsum);
    vars = (n * (n - 1) * (2 * n + 5) - tpsum) / 18.0;
}

Console.WriteLine(vars);
